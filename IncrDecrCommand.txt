package Commands;

import java.util.Map;

public class IncrDecrCommand {

    /* Cette méthode prend en entrée une Map, une chaîne de caractères représentant la clé, et une valeur 
    numérique entière. Si la clé existe dans la map, la valeur associée est modifiée en ajoutant la valeur
    numérique donnée en entrée. Si la clé n'existe pas, une nouvelle clé est créée avec la valeur donnée
    en entrée. Si la valeur donnée en entrée est négative, la valeur associée à la clé est décrémentée.
    La méthode retourne la map modifiée. 
    */
    public static Map<String, String> IncrDecr(Map<String, String> map, String key, int value) {
        if (map.containsKey(key)) {
            int oldValue = Integer.parseInt(map.get(key));
            int newValue = oldValue + value;
            map.put(key, String.valueOf(newValue));
        } else {
            map.put(key, String.valueOf(value));
        }
        return map;
    }
}
